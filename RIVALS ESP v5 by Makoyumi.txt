local function HexToColor3(hex)
    hex = hex:gsub("#", ""):gsub(" ", "")
    if #hex ~= 6 then return Color3.fromRGB(255, 105, 180) end
    local r = tonumber(hex:sub(1,2), 16) / 255
    local g = tonumber(hex:sub(3,4), 16) / 255
    local b = tonumber(hex:sub(5,6), 16) / 255
    if not r or not g or not b then return Color3.fromRGB(255, 105, 180) end
    return Color3.new(r, g, b)
end

local function SaveColor(hexStr)
    pcall(function()
        if writefile then writefile(CONFIG_COLOR, hexStr) end
    end)
end

local function LoadSavedColor()
    local ok, data = pcall(function()
        if isfile and isfile(CONFIG_COLOR) then return readfile(CONFIG_COLOR) end
    end)
    if ok and data and type(data) == "string" and #data > 3 then
        return HexToColor3(data)
    end
    return nil
end

local function SaveToggles()
    pcall(function()
        if writefile then
            local str = string.format("%d,%d,%d,%d,%d,%d,%.2f,%d,%d,%d",
                ShowBox and 1 or 0,
                ShowTracer and 1 or 0,
                ShowName and 1 or 0,
                ShowHealth and 1 or 0,
                RainbowESP and 1 or 0,
                ShowDistance and 1 or 0,
                RainbowSpeed,
                ShowHeadDot and 1 or 0,
                ShowCornerBox and 1 or 0,
                TeamCheck and 1 or 0)
            writefile(CONFIG_TOGGLES, str)
        end
    end)
end

local function LoadToggles()
    local ok, data = pcall(function()
        if isfile and isfile(CONFIG_TOGGLES) then
            return readfile(CONFIG_TOGGLES)
        end
    end)
    if ok and data and type(data) == "string" then
        local t = {}
        for num in data:gmatch("[^,]+") do table.insert(t, num) end
        if #t >= 6 then
            ShowBox = (t[1] == "1")
            ShowTracer = (t[2] == "1")
            ShowName = (t[3] == "1")
            ShowHealth = (t[4] == "1")
            RainbowESP = (t[5] == "1")
            ShowDistance = (t[6] == "1")
            if #t >= 7 then RainbowSpeed = tonumber(t[7]) or 0.6 end
            if #t >= 8 then ShowHeadDot = (t[8] == "1") end
            if #t >= 9 then ShowCornerBox = (t[9] == "1") end
            if #t >= 10 then TeamCheck = (t[10] == "1") end
        end
    end
    SaveToggles()
end

-- Team Check function
local function IsTeammate(player)
    if player == LocalPlayer then return true end
    if not TeamCheck then return false end
    
    local char = player.Character
    if not char then return false end
    
    local head = char:FindFirstChild("Head")
    if head then
        for _, obj in ipairs(head:GetDescendants()) do
            if obj:IsA("BillboardGui") then
                return true
            end
        end
    end
    
    for _, obj in ipairs(char:GetDescendants()) do
        if obj:IsA("BillboardGui") then
            return true
        end
    end
    
    if player.Team and LocalPlayer.Team and player.Team == LocalPlayer.Team then
        return true
    end
    
    return false
end

local function UpdateESPColors(newPink)
    Pink = newPink
    PinkLight = Color3.new(
        math.min(1, newPink.R + 0.25),
        math.min(1, newPink.G + 0.25),
        math.min(1, newPink.B + 0.25)
    )
    for _, esp in pairs(ESPObjects) do
        if esp.Box then esp.Box.Color = PinkLight end
        if esp.Tracer then esp.Tracer.Color = Pink end
        if esp.Name then esp.Name.Color = Pink end
        if esp.Distance then esp.Distance.Color = Pink end
    end
end

        for _, obj in pairs(esp) do
            if obj.Remove then obj:Remove() end

RunService.RenderStepped:Connect(function()

    local rainbowLight = RainbowESP and rainbowColor or PinkLight

    -- Rainbow menu accents
    local accentColor = RainbowESP and rainbowColor or Pink
    pcall(function()
        if menuStroke then menuStroke.Color = accentColor end
        if title then title.TextColor3 = accentColor end
        if applyHexBtn then applyHexBtn.BackgroundColor3 = accentColor end
        if previewStroke then previewStroke.Color = accentColor end

            if activeTab == "colors" then
                colorsTab.BackgroundColor3 = accentColor
            elseif activeTab == "toggles" then
                togglesTab.BackgroundColor3 = accentColor
            elseif activeTab == "config" then
                configTab.BackgroundColor3 = accentColor
            end
        end
        if speedFill then speedFill.BackgroundColor3 = accentColor end
    end)

    for player, esp in pairs(ESPObjects) do
        local char = player.Character
        if char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Humanoid") then
            local root = char.HumanoidRootPart
            local hum = char.Humanoid
            
            if hum.Health > 0 then
                if TeamCheck and IsTeammate(player) then
                    esp.Box.Visible = false
                    esp.Tracer.Visible = false
                    esp.HealthBG.Visible = false
                    esp.HealthFill.Visible = false
                    esp.Name.Visible = false
                    esp.Distance.Visible = false
                    esp.HeadDot.Visible = false
                    for i = 1, 8 do if esp.CornerLines[i] then esp.CornerLines[i].Visible = false end end
                    continue
                end
                
                local rootPos, onScreen = Camera:WorldToViewportPoint(root.Position)
                
                if onScreen then
                    local head = char:FindFirstChild("Head")
                    local headPos = head and Camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.8, 0)) or rootPos
                    local headDotPos = head and Camera:WorldToViewportPoint(head.Position) or rootPos
                    local feetPos = Camera:WorldToViewportPoint(root.Position - Vector3.new(0, 3, 0))
                    
                    local boxHeight = math.max(10, math.abs(headPos.Y - feetPos.Y))
                    local boxWidth = boxHeight * 0.6
                    
                    esp.Box.Color = rainbowLight
                    esp.Box.Size = Vector2.new(boxWidth, boxHeight)
                    esp.Box.Position = Vector2.new(rootPos.X - boxWidth/2, headPos.Y)
                    esp.Box.Visible = ShowBox and not ShowCornerBox
                    
                    if ShowCornerBox then
                        local topLeft = Vector2.new(rootPos.X - boxWidth/2, headPos.Y)
                        local topRight = Vector2.new(rootPos.X + boxWidth/2, headPos.Y)
                        local bottomLeft = Vector2.new(rootPos.X - boxWidth/2, headPos.Y + boxHeight)
                        local bottomRight = Vector2.new(rootPos.X + boxWidth/2, headPos.Y + boxHeight)
                        local cLen = 7

                        esp.CornerLines[1].From = topLeft
                        esp.CornerLines[1].To = topLeft + Vector2.new(cLen, 0)
                        esp.CornerLines[2].From = topLeft
                        esp.CornerLines[2].To = topLeft + Vector2.new(0, cLen)

                        esp.CornerLines[3].From = topRight
                        esp.CornerLines[3].To = topRight + Vector2.new(-cLen, 0)
                        esp.CornerLines[4].From = topRight
                        esp.CornerLines[4].To = topRight + Vector2.new(0, cLen)

                        esp.CornerLines[5].From = bottomLeft
                        esp.CornerLines[5].To = bottomLeft + Vector2.new(cLen, 0)
                        esp.CornerLines[6].From = bottomLeft
                        esp.CornerLines[6].To = bottomLeft + Vector2.new(0, -cLen)

local function CreateColorSlider(parent, yPos, labelText, initialVal, onValueChanged)
    local sliderFrame = Instance.new("Frame")
    sliderFrame.Parent = parent
    sliderFrame.Position = UDim2.new(0, 15, 0, yPos)
    sliderFrame.Size = UDim2.new(1, -30, 0, 42)

    sLabel.BackgroundTransparency = 1
    sLabel.Text = labelText
    sLabel.TextColor3 = Color3.fromRGB(220, 220, 220)
    sLabel.Font = Enum.Font.GothamBold
    sLabel.TextSize = 14
    
    local valLabel = Instance.new("TextLabel")
    valLabel.Parent = sliderFrame
    valLabel.Position = UDim2.new(0.78, 0, 0, 0)
    valLabel.Size = UDim2.new(0.22, 0, 0, 18)
    valLabel.BackgroundTransparency = 1
    valLabel.Text = tostring(initialVal)
    valLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    valLabel.Font = Enum.Font.GothamBold
    valLabel.TextSize = 14
    valLabel.TextXAlignment = Enum.TextXAlignment.Right
    
    local bg = Instance.new("Frame")
    bg.Parent = sliderFrame
    bg.Position = UDim2.new(0, 0, 0, 22)
    bg.Size = UDim2.new(1, 0, 0, 10)
    bg.BackgroundColor3 = Color3.fromRGB(45, 45, 55)
    local bgCorner = Instance.new("UICorner")
    bgCorner.CornerRadius = UDim.new(0, 5)
    bgCorner.Parent = bg
    
    local fillColor = (labelText == "Red") and Color3.fromRGB(255, 90, 90) or
                      (labelText == "Green") and Color3.fromRGB(90, 255, 90) or
                      Color3.fromRGB(90, 140, 255)
    local fill = Instance.new("Frame")
    fill.Parent = bg
    fill.Size = UDim2.new(initialVal / 255, 0, 1, 0)
    fill.BackgroundColor3 = fillColor
    local fillCorner = Instance.new("UICorner")
    fillCorner.CornerRadius = UDim.new(0, 5)
    fillCorner.Parent = fill
    
    local knob = Instance.new("Frame")
    knob.Parent = bg
    knob.Size = UDim2.new(0, 18, 0, 18)
    knob.Position = UDim2.new(initialVal / 255, -9, 0.5, -9)
    knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    knob.BorderSizePixel = 0
    local knobCorner = Instance.new("UICorner")
    knobCorner.CornerRadius = UDim.new(1, 0)
    knobCorner.Parent = knob
    
    local dragging = false
    local function updateSlider(val)
        val = math.clamp(math.floor(val + 0.5), 0, 255)
        valLabel.Text = tostring(val)
        fill.Size = UDim2.new(val / 255, 0, 1, 0)
        knob.Position = UDim2.new(val / 255, -9, 0.5, -9)
        onValueChanged(val)
    end
    
    knob.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
        end
    end)
    bg.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            local mouseX = input.Position.X
            local bgPos = bg.AbsolutePosition.X
            local bgSize = bg.AbsoluteSize.X
            local relative = (mouseX - bgPos) / bgSize
            updateSlider(relative * 255)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local mouseX = input.Position.X
            local bgPos = bg.AbsolutePosition.X

        end
    end)

end

-- Keybinds: Right Shift + Insert
UserInputService.InputBegan:Connect(function(input, gp)
    if gp then return end
    if input.KeyCode == Enum.KeyCode.Insert then
        Enabled = not Enabled
        if not Enabled then
            for _, esp in pairs(ESPObjects) do
                for _, obj in pairs(esp) do obj.Visible = false end
            end
        end
    elseif input.KeyCode == Enum.KeyCode.RightShift then
        menuGui.Enabled = not menuGui.Enabled
        if menuGui.Enabled then
            UserInputService.MouseBehavior = Enum.MouseBehavior.Default
            UserInputService.MouseIconEnabled = true
        else
            UserInputService.MouseBehavior = Enum.MouseBehavior.LockCenter
        end
    end
end)

RunService.RenderStepped:Connect(function()
    if menuGui.Enabled then
        UserInputService.MouseBehavior = Enum.MouseBehavior.Default
        UserInputService.MouseIconEnabled = true
    end
end)

-- Rainbow menu accent loop
RunService.RenderStepped:Connect(function()
    if not RainbowESP then return end
    local rainbowCol = GetRainbowColor()
    pcall(function()
        if menuStroke then menuStroke.Color = rainbowCol end
        if title then title.TextColor3 = rainbowCol end
        if applyHexBtn then applyHexBtn.BackgroundColor3 = rainbowCol end
        if previewStroke then previewStroke.Color = rainbowCol end
        if colorsTab and togglesTab and configTab then
            if activeTab == "colors" then
                colorsTab.BackgroundColor3 = rainbowCol
            elseif activeTab == "toggles" then
                togglesTab.BackgroundColor3 = rainbowCol
            elseif activeTab == "config" then
                configTab.BackgroundColor3 = rainbowCol
            end
        end
        if speedFill then speedFill.BackgroundColor3 = rainbowCol end
    end)
end)
